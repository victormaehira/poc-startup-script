sudo mkdir -p /opt/projetos
cd /opt/projetos

sudo apt -y install git
sudo git clone https://github.com/victormaehira/poc-startup-script.git
cd poc-startup-script

#escolha dir = biozoomserver ou quality ou facecheck
export dir=quality

echo "" >> ./envlist-repo/facetec-env-list-dev.txt
echo "SPRING_DATASOURCE_USERNAME=<COLOQUE_USER_AQUI>" >> ./envlist-repo/facetec-env-list-dev.txt
echo "SPRING_DATASOURCE_PASSWORD=<COLOQUE_A_SENHA_AQUI>" >> ./envlist-repo/facetec-env-list-dev.txt
echo "SPRING_DATASOURCE_URL=<COLOQUE_A_URL_AQUI>" >> ./envlist-repo/facetec-env-list-dev.txt

echo "" >> ./envlist-repo/facecheck-env-list-dev.txt
echo "SPRING_DATASOURCE_USERNAME=<COLOQUE_USER_AQUI>" >> ./envlist-repo/facecheck-env-list-dev.txt
echo "SPRING_DATASOURCE_PASSWORD=<COLOQUE_A_SENHA_AQUI_AQUI>" >> ./envlist-repo/facecheck-env-list-dev.txt
echo "SPRING_DATASOURCE_URL=<COLOQUE_A_URL_AQUI>" >> ./envlist-repo/facecheck-env-list-dev.txt

source .env

sudo apt -y install docker.io

sudo curl -L "https://github.com/docker/compose/releases/download/1.29.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

sudo docker network create facecheck_network

sudo gcloud auth configure-docker
sudo cp ${dir}/docker-compose.yml ./
sudo docker-compose up -d
